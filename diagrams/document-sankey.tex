\newcommand{\flowband}[7]{%
  \path[fill=#7, draw=black!18, line width=0.35pt]
    (#1,#2) .. controls (#1 + 1.1,#2) and (#4 - 1.1,#5) .. (#4,#5)
    -- (#4,#6)
    .. controls (#4 - 1.1,#6) and (#1 + 1.1,#3) .. (#1,#3)
    -- cycle;
}
